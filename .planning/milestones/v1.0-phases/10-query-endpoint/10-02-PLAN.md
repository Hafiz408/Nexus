```python
"""Tests for POST /query SSE endpoint (API-03, API-04).

Uses FastAPI TestClient with stream=True to consume the SSE body.
All external I/O is monkeypatched:
  - app.api.query_router.get_status   → returns IndexStatus or None
  - app.api.query_router.load_graph   → returns nx.DiGraph()
  - app.api.query_router.graph_rag_retrieve → returns (nodes, stats)
  - app.api.query_router.explore_stream     → async generator of token strings
"""
import asyncio
import json

import networkx as nx
import pytest
from fastapi.testclient import TestClient

from app.main import app
from app.models.schemas import CodeNode, IndexStatus

# Shared fixtures

@pytest.fixture()
def client():
    with TestClient(app) as c:
        yield c

@pytest.fixture()
def sample_node():
    return CodeNode(
        node_id="src/foo.py::bar",
        name="bar",
        type="function",
        file_path="src/foo.py",
        line_start=10,
        line_end=20,
        signature="def bar():",
        docstring="Does bar.",
        body_preview="return 42",
    )

@pytest.fixture()
def sample_stats():
    return {"seed_nodes": 1, "expanded_nodes": 3, "returned_nodes": 1}

def _complete_status():
    return IndexStatus(status="complete", nodes_indexed=5, edges_indexed=2)

def _make_async_gen(*tokens):
    """Return an async generator that yields the given token strings."""
    async def _gen(nodes, question, **kwargs):
        for t in tokens:
            yield t
    return _gen
```

**Monkeypatching pattern** (match the namespace where the symbols are imported, not origin modules — consistent with pipeline tests):

```python
def test_happy_path_yields_token_events(client, monkeypatch, sample_node, sample_stats):
    monkeypatch.setattr("app.api.query_router.get_status", lambda repo_path: _complete_status())
    monkeypatch.setattr("app.api.query_router.load_graph", lambda repo_path: nx.DiGraph())
    monkeypatch.setattr(
        "app.api.query_router.graph_rag_retrieve",
        lambda question, repo_path, G, max_nodes, hop_depth: ([sample_node], sample_stats),
    )
    monkeypatch.setattr(
        "app.api.query_router.explore_stream",
        _make_async_gen("Hello", " world"),
    )
    with client.stream("POST", "/query", json={"question": "What is bar?", "repo_path": "/repo"}) as r:
        body = r.text
    assert "event: token" in body
    token_events = [line for line in body.splitlines() if line.startswith("event: token")]
    assert len(token_events) == 2
```

**Important notes:**
- Use `client.stream("POST", ...)` (not `client.post(...)`) to consume SSE response body correctly in TestClient
- `asyncio.to_thread` wraps synchronous calls in the router — monkeypatched synchronous mocks work correctly (to_thread calls any callable in a thread pool)
- `explore_stream` is an async generator; the mock must be an async generator function (use `_make_async_gen` pattern above)
- Do NOT use `pytest.mark.asyncio` — consistent with `test_pipeline.py` and `test_explorer.py` patterns (asyncio.run or TestClient handles event loop)
- `app.state.graph_cache` is initialized in lifespan; `TestClient` as context manager (`with TestClient(app) as c`) triggers lifespan, so the cache dict will exist
