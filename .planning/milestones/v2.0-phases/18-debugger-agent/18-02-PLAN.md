```python
@pytest.fixture
def mock_llm_factory():
    """Patch get_llm at source module — lazy import requires source-level patch.

    The debug() function does: chain = prompt | llm; response = chain.invoke(...)
    So we mock: llm.__or__ returns a mock chain; chain.invoke returns a mock response.
    """
    with patch("app.core.model_factory.get_llm") as mock_factory:
        mock_response = MagicMock()
        mock_response.content = "The bug likely originates in hop2a due to missing error handling and high complexity."

        mock_chain = MagicMock()
        mock_chain.invoke.return_value = mock_response

        mock_llm = MagicMock()
        mock_llm.__or__ = MagicMock(return_value=mock_chain)

        mock_factory.return_value = mock_llm
        yield mock_factory
```

Note: If during implementation the chain is invoked differently (e.g. chain is callable not using .invoke), adjust accordingly. The research file notes: "The exact mock setup for LCEL chain (`prompt | llm`) depends on how `debug()` invokes the chain." Check the actual `debug()` implementation's chain invocation pattern and mirror it here.

**Test 8 — Impact radius is correct (DBUG-03):**
```python
def test_impact_radius_correct(mock_llm_factory, debug_graph):
    result = debug("error in hop2a", debug_graph)
    if not result.suspects:
        pytest.skip("no suspects — graph topology may not score hop2a as top")
    top_suspect_id = result.suspects[0].node_id
    # compute expected callers from debug_graph directly
    expected_callers = {
        pred for pred in debug_graph.predecessors(top_suspect_id)
        if debug_graph.edges[pred, top_suspect_id].get("type") == "CALLS"
    }
    assert set(result.impact_radius) == expected_callers
```
